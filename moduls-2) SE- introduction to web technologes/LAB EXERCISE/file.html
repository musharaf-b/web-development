C Program: File Create, Write & Read

#include <stdio.h>

int main() {
    FILE *fp;
    char text[] = "Hello, this is a file handling example!";
    char ch;

    // Create and write to file
    fp = fopen("sample.txt", "w");
    fprintf(fp, "%s", text);
    fclose(fp);

    // Read from file
    fp = fopen("sample.txt", "r");
    printf("File contents:\n");

    while((ch = fgetc(fp)) != EOF) {
        printf("%c", ch);
    }

    fclose(fp);

    return 0;
}

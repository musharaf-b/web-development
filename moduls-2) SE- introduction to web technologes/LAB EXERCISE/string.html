C Program: strcat() and strlen()

#include <stdio.h>
#include <string.h>

int main() {
    char str1[100], str2[100];

    printf("Enter first string: ");
    scanf("%s", str1);

    printf("Enter second string: ");
    scanf("%s", str2);

    // Concatenate strings
    strcat(str1, str2);

    // Display result
    printf("Concatenated string: %s\n", str1);

    // Find length
    printf("Length of concatenated string: %lu", strlen(str1));

    return 0;
}

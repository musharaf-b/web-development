C Program: 1D & 2D Arrays

#include <stdio.h>

int main() {
    int i, j, sum = 0;

    // One-dimensional array
    int arr[5] = {1, 2, 3, 4, 5};

    printf("One-Dimensional Array:\n");
    for(i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }

    // Two-dimensional array (3x3)
    int matrix[3][3] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };

    printf("\n\nTwo-Dimensional Array:\n");
    for(i = 0; i < 3; i++) {
        for(j = 0; j < 3; j++) {
            printf("%d ", matrix[i][j]);
            sum += matrix[i][j];
        }
        printf("\n");
    }

    printf("\nSum of all elements: %d", sum);

    return 0;
}
